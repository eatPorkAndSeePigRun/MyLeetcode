# 114. Flatten Binary Tree to Linked List

## Description:

Given a binary tree, flatten it to a linked list in-place.

## Solution:

```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void flatten(TreeNode* root) {
        if (!root)  return;
        
        flatten(root->left);
        flatten(root->right);
        
        auto tmp = root->right;
        root->right = root->left;
        root->left = nullptr;
        while (root->right)
            root = root->right;
        root->right = tmp;
    }
};
```
