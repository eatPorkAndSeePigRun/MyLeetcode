# 394. Decode String

## Description:

Given an encoded string, return its decoded string.

## Solution:

```c++
class Solution {
public:
    string decodeString(string s) {
        stack<int> num_stack;
        stack<string> str_stack;
        int num = 0;
        string str;
        
        for (int i = 0; i < s.size(); ++i) {
            if (isdigit(s[i])) {
                num = 10 * num + s[i] - '0';
            } else if (isalpha(s[i])) {
                str += s[i];
            } else if (s[i] == '[') {
                num_stack.push(num);
                str_stack.push(str);
                num = 0;
                str = "";
            } else if (s[i] == ']') {
                int count = num_stack.top();
                num_stack.pop();
                for (int j = 0; j < count; ++j)
                    str_stack.top() += str;
                str = str_stack.top();
                str_stack.pop();
            }
        }
        
        return str;
    }
};
```
