# 19. Remove Nth Node From End of List

## Description:

Given a linked list, remove the n-th node from the end of list and return its head.

## Solution:

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        if (!head)  return nullptr;
        auto p = head;
        
        for (auto i = 0; i < n; ++i) {
            if (p)
                p = p->next;
            else
                return nullptr;
        }
        
        auto q = head;
        ListNode *pre = nullptr;
        while (p) {
            pre = q;
            q = q->next;
            p = p->next;
        }
        
        if (q == head) {
            auto ans = head->next;
            delete head;
            head = nullptr;
            return ans;
        }
        
        pre->next = q->next;
        delete q;
        q = nullptr;
        
        return head;
    }
};
```
