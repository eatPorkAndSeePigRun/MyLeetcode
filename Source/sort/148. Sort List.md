# 148. Sort List

## Description:

Sort a linked list in O(n log n) time using constant space complexity.

## Solution:

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        quickSort(head, nullptr);
        return head;
    }
    
    void quickSort(ListNode *start, ListNode *end) {
        if (!start || start->next == end)
            return;
        
        auto key = start->val;
        auto i = start;
        auto j = start;
        while (j && j != end) {
            if (j->val < key) {
                swap(i->val, j->val);
                i = i->next;
            }
            j = j->next;
        }
        if (i->val == key)
            i = i->next;
        
        quickSort(start, i);
        quickSort(i, end);
    }
};
```

remark：

- 快排，归并
